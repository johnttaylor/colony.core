99 Result_T result;
100 result = writeToDevice( device7BitAddress, sizeof( reg ), &reg, true );
101 if ( result == eSUCCESS )

103 result = readFromDevice( device7BitAddress, sizeof( dstReadResult ), &dstReadResult );
